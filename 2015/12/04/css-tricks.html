  1. Repeating class to increase priority
<div class="foo">
  <div class="bar">Minh Quy</div>
</div>

.foo .bar {
  color: red;
}

.bar {
  color: green;
}

// => red

.foo .bar {
  color: red;
}

.bar.bar {
  color: green;
}

// => green
